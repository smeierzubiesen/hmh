25 internal void Win32RenderGradient(win32_offscreen_buffer Buffer, int XOffset, int YOffset) {
26  uint8 *Row = (uint8 *)Buffer.Memory;
27  for (int Y = 0; Y < Buffer.Height; ++Y) {
28  uint32 *Pixel = (uint32 *)Row;
29  for (int X = 0; X < Buffer.Width; ++X) {
30  uint8 Blue = (X + XOffset);
31  uint8 Green = (Y + YOffset);
32  uint8 Red = (X + XOffset);
33  *Pixel++ = ((Red << 16) | (Green << 8) | (Blue));
34  }
35  Row += Buffer.Pitch;
36  }
